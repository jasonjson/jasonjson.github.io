Given an integer matrix, find a submatrix where the sum of numbers is zero. Your code should return the coordinate of the left-up and right-down number.

[expand title="code"]

public class Solution {
    /**
     * @param matrix an integer matrix
     * @return the coordinate of the left-up and right-down number
     */
    public static int[][] submatrixSum(int[][] matrix) {
        // Write your code here
        if (matrix == null || matrix.length == 0) return new int[2][2];

        int[][] sum = new int[matrix.length + 1][matrix[0].length + 1];
        for (int i = 1; i <= matrix.length; i++) {
            for (int j = 1; j <= matrix[0].length; j++) {
                sum[i][j] = matrix[i-1][j-1] + sum[i-1][j] + sum[i][j-1] - sum[i-1][j-1];
            }
        }//最好matrix.length + 1,因为要包括以(0,0)为右下角的矩阵
        for (int i = 0; i + 1 <= matrix.length; i++) {
            for (int j = i + 1; j <= matrix.length; j++) {
                //the following is like a 1D problem
                HashMap map = new HashMap<>();
                for (int k = 0; k <= matrix[0].length; k++) {
                    int diff = sum[j][k] - sum[i][k];//sum between two matrixes
                    if (map.containsKey(diff)) {
                        return new int[][] {{i, map.get(diff)}, {j-1, k - 1}};
                    } else {
                        map.put(diff, k);
                    }
                }
            }
        }
        return new int[2][2];
    }
}
